// Create Assign instruction
const assignAccounts = [
  { address: accountToDelegate, role: AccountRole.WRITABLE_SIGNER },
  { address: SYSTEM_PROGRAM_ADDRESS, role: AccountRole.READONLY },
];
const discriminator = Buffer.alloc(4);
discriminator.writeUInt32LE(1, 0); // SystemInstruction.Assign = 1
const addressEncoder = getAddressEncoder();
const assignInstructionData = Buffer.concat([
  discriminator,
  addressEncoder.encode(DELEGATION_PROGRAM_ID),
]); // Concatenate discriminator + new owner pubkey
const assignIx: Instruction = {
  accounts: assignAccounts,
  programAddress: SYSTEM_PROGRAM_ADDRESS,
  data: assignInstructionData,
};

// Create Delegate instruction
const delegateBufferPda =
  await delegateBufferPdaFromDelegatedAccountAndOwnerProgram(
    accountToDelegate,
    SYSTEM_PROGRAM_ADDRESS
  );
const delegationRecordPda = await delegationRecordPdaFromDelegatedAccount(
  accountToDelegate
);
const delegationMetadataPda = await delegationMetadataPdaFromDelegatedAccount(
  accountToDelegate
);

const delegateAccounts = [
  { address: userPubkey, role: AccountRole.WRITABLE_SIGNER },
  { address: SYSTEM_PROGRAM_ADDRESS, role: AccountRole.READONLY },
  { address: accountToDelegate, role: AccountRole.WRITABLE_SIGNER },
  { address: SYSTEM_PROGRAM_ADDRESS, role: AccountRole.READONLY },
  { address: delegateBufferPda, role: AccountRole.WRITABLE },
  { address: delegationRecordPda, role: AccountRole.WRITABLE },
  { address: delegationMetadataPda, role: AccountRole.WRITABLE },
  { address: DELEGATION_PROGRAM_ID, role: AccountRole.READONLY },
];

const delegateInstructionData = Buffer.from([0, 0, 0, 0, 0, 0, 0, 0]);

const delegateIx: Instruction = {
  accounts: delegateAccounts,
  programAddress: DELEGATION_PROGRAM_ID,
  data: delegateInstructionData,
};

// Create transaction message with both instructions
const transactionMessage = pipe(
  createTransactionMessage({ version: 0 }),
  (tx) => setTransactionMessageFeePayer(userPubkey, tx),
  (tx) => appendTransactionMessageInstructions([assignIx, delegateIx], tx)
);
