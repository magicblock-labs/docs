use ephemeral_rollups_pinocchio::types::{
    Member,
    MemberFlags,
    MembersArgs,

// Close permission by clearing all members
// IMPORTANT: The permission must be undelegated to Solana first

    members: None,
};

// Pass members_args with None to close permission
// This removes all members and closes the permission account
